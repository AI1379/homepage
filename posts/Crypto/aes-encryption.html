void aes::InvMixColumn() {
    const byte y[16] = {0x0e, 0x0b, 0x0d, 0x09,
                        0x09, 0x0e, 0x0b, 0x0d,
                        0x0d, 0x09, 0x0e, 0x0b,
                        0x0b, 0x0d, 0x09, 0x0e};
    byte arr[4];
    for (int i = 0; i < Nb; ++i) {
        for (int k = 0; k < 4; ++k) {
            arr[k] = 0;
            for (int j = 0; j < 4; ++j) {
                arr[k] = arr[k] ^ GFMul(y[k * 4 + j], state[i * 4 + j]);
            }
        }
        for (int k = 0; k < 4; ++k) {
            state[i * 4 + k] = arr[k];
        }
    }
}

void aes::InvShiftRow() {
    for (int i = 1; i < Nb; i++) {
        for (int j = 0; j < Nb - i; ++j) {
            byte tmp = state[i];
            state[i + 0 * 4] = state[i + (0 + 1) * 4];
            state[i + 1 * 4] = state[i + (1 + 1) * 4];
            state[i + 2 * 4] = state[i + (2 + 1) * 4];
            state[i + 3 * 4] = tmp;
        }
    }
}
